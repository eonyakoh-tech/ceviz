import torch

class MultiModalImageAnalyzer(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        
        # 비전 인코더
        self.vision_encoder = VisionEncoder(config.vision)
        
        # 텍스트 인코더
        self.text_encoder = TextEncoder(config.text)
        
        # 크로스 모달 퓨전
        self.fusion_layer = CrossModalFusion(config.fusion)
        
        # 태스크별 헤드
        self.classification_head = ClassificationHead(config.num_classes)
        self.captioning_head = CaptioningHead(config.vocab_size)
        self.vqa_head = VQAHead(config.num_answers)
        
    def forward(self, images, text=None, task='classification'):
        # 비전 특성 추출
        vision_features = self.vision_encoder(images)
        
        if text is not None:
            # 텍스트 특성 추출
            text_features = self.text_encoder(text)
            
            # 크로스 모달 퓨전
            fused_features = self.fusion_layer(vision_features, text_features)
        else:
            fused_features = vision_features
            
        # 태스크별 출력
        if task == 'classification':
            return self.classification_head(fused_features)
        elif task == 'captioning':
            return self.captioning_head(fused_features)
        elif task == 'vqa':
            return self.vqa_head(fused_features)
```

## 2. 비전 인코더 모듈

```python
class VisionEncoder(nn.Module):
    def __init__(self, config):
        super().__init__()
        
        # 백본 네트워크 (ResNet, ViT, EfficientNet 등)
        if config.backbone == 'resnet':
            self.backbone = models.resnet50(pretrained=True)
            self.backbone.fc = nn.Identity()
            self.feature_dim = 2048
            
        elif config.backbone == 'vit':
            self.backbone = models.vit_b_16(pretrained=True)
            self.backbone.heads = nn.Identity()
            self.feature_dim = 768
            
        elif config.backbone == 'efficientnet':
            self.backbone = models.efficientnet_b0(pretrained=True)
            self.backbone.classifier = nn.Identity()
            self.feature_dim = 1280
            
        # 특성 투영 레이어
        self.projection = nn.Sequential(
            nn.Linear(self.feature_dim, config.hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(config.hidden_dim, config.output_dim)
        )
        
        # 어텐션 메커니즘
        self.attention = SpatialAttention(config.output_dim)
        
    def forward(self, x):
        # 백본을 통한 특성 추출
        features = self.backbone(x)
        
        # 특성 투영
        projected = self.projection(features)
        
        # 어텐션 적용
        attended_features = self.attention(projected)
        
        return attended_features

class SpatialAttention(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.attention = nn.MultiheadAttention(dim, num_heads=8, batch_first=True)
        self.norm = nn.LayerNorm(dim)
        
    def forward(self, x):
        # Self-attention
        attended, _ = self.attention(x, x, x)
        return self.norm(attended + x)
```

## 3. 텍스트 인코더 모듈

```python
from transformers import AutoModel, AutoTokenizer

class TextEncoder(nn.Module):
    def __init__(self, config):
        super().__init__()
        
        # 사전 훈련된 언어 모델
        self.model_name = config.model_name  # 'bert-base-uncased', 'roberta-base' 등
        self.encoder = AutoModel.from_pretrained(self.model_name)
        self.tokenizer = AutoTokenizer.from_pretrained(self.model_name)
        
        # 투영 레이어
        self.projection = nn.Sequential(
            nn.Linear(self.encoder.config.hidden_size, config.hidden_dim),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(config.hidden_dim, config.output_dim)
        )
        
    def forward(self, text_inputs):
        # 텍스트 인코딩
        if isinstance(text_inputs, list):
            # 토큰화
            encoded = self.tokenizer(
                text_inputs, 
                padding=True, 
                truncation=True, 
                return_tensors='pt',
                max_length=512
            )
            
            # BERT/RoBERTa를 통한 인코딩
            outputs = self.encoder(**encoded)
            text_features = outputs.last_hidden_state[:, 0]  # [CLS] 토큰
        else:
            text_features = text_inputs
            
        # 투영
        projected = self.projection(text_features)
        
        return projected
```

class CrossModalFusion(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.fusion_type = config.fusion_type
        self.hidden_dim = config.hidden_dim
        
        if self.fusion_type == 'attention':
            self.cross_attention = CrossModalAttention(config)
        elif self.fusion_type == 'transformer':
            self.fusion_transformer = FusionTransformer(config)
        elif self.fusion_type == 'bilinear':
            self.bilinear_fusion = BilinearFusion(config)
            
    def forward(self, vision_features, text_features):
        if self.fusion_type == 'attention':
            return self.cross_attention(vision_features, text_features)
        elif self.fusion_type == 'transformer':
            return self.fusion_transformer(vision_features, text_features)
        elif self.fusion_type == 'bilinear':
            return self.bilinear_fusion(vision_features, text_features)
        elif self.fusion_type == 'concat':
            return torch.cat([vision_features, text_features], dim=-1)

class CrossModalAttention(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.dim = config.hidden_dim
        
        # 크로스 어텐션 레이어
        self.v_to_t_attention = nn.MultiheadAttention(
            self.dim, num_heads=8, batch_first=True
        )
        self.t_to_v_attention = nn.MultiheadAttention(
            self.dim, num_heads=8, batch_first=True
        )
        
        # 정규화 레이어
        self.v_norm = nn.LayerNorm(self.dim)
        self.t_norm = nn.LayerNorm(self.dim)
        
        # 피드포워드 네트워크
        self.ffn = nn.Sequential(
            nn.Linear(self.dim * 2, self.dim * 4),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(self.dim * 4, self.dim)
        )
        
    def forward(self, vision_features, text_features):
        # 크로스 어텐션
        v_attended, _ = self.v_to_t_attention(
            vision_features, text_features, text_features
        )
        t_attended, _ = self.t_to_v_attention(
            text_features, vision_features, vision_features
        )
        
        # 정규화
        v_attended = self.v_norm(v_attended + vision_features)
        t_attended = self.t_norm(t_attended + text_features)
        
        # 융합
        fused = torch.cat([v_attended, t_attended], dim=-1)
        output = self.ffn(fused)
        
        return output

class BilinearFusion(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.dim_v = config.vision_dim
        self.dim_t = config.text_dim
        self.dim_h = config.hidden_dim
        
        # 바이리니어 풀링
        self.bilinear = nn.Bilinear(self.dim_v, self.dim_t, self.dim_h)
        
        # 개별 변환
        self.v_transform = nn.Linear(self.dim_v, self.dim_h)
        self.t_transform = nn.Linear(self.dim_t, self.dim_h)
        
        # 출력 레이어
        self.output = nn.Sequential(
            nn.Linear(self.dim_h * 3, self.dim_h),
            nn.ReLU(),
            nn.Dropout(0.1)
        )
        
    def forward(self, vision_features, text_features):
        # 바이리니어 융합
        bilinear_out = self.bilinear(vision_features, text_features)
        
        # 개별 변환
        v_out = self.v_transform(vision_features)
        t_out = self.t_transform(text_features)
        
        # 결합 및 출력
        combined = torch.cat([bilinear_out, v_out, t_out], dim=-1)
        output = self.output(combined)
        
        return output
```

## 5. 태스크별 헤드

```python
class ClassificationHead(nn.Module):
    def __init__(self, input_dim, num_classes):
        super().__init__()
        self.classifier = nn.Sequential(
            nn.Linear(input_dim, input_dim // 2),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(input_dim // 2, num_classes)
        )
        
    def forward(self, features):
        return self.classifier(features)

class CaptioningHead(nn.Module):
    def __init__(self, input_dim, vocab_size, max_length=100):
        super().__init__()
        self.vocab_size = vocab_size
        self.max_length = max_length
        
        # LSTM 디코더
        self.decoder = nn.LSTM(
            input_size=input_dim,
            hidden_size=512,
            num_layers=2,
            batch_first=True
        )
        
        # 출력 레이어
        self.output_proj = nn.Linear(512, vocab_size)
        
        # 어텐션
        self.attention = nn.MultiheadAttention(512, num_heads=8, batch_first=True)
        
    def forward(self, features, target_captions=None):
        if target_captions is not None:
            # 훈련 모드
            decoder_out, _ = self.decoder(features)
            output = self.output_proj(decoder_out)
            return output
        else:
            # 추론 모드
            return self.generate(features)
    
    def generate(self, features):
        batch_size = features.size(0)
        generated = []
        
        # 자동회귀 생성
        hidden = None
        current_input = features
        
        for _ in range(self.max_length):
            output, hidden = self.decoder(current_input, hidden)
            logits = self.output_proj(output)
            predicted = torch.argmax(logits, dim=-1)
            generated.append(predicted)
            
            # 다음 입력 준비
            current_input = self.embedding(predicted)
            
        return torch.stack(generated, dim=1)

class VQAHead(nn.Module):
    def __init__(self, input_dim, num_answers):
        super().__init__()
        self.answer_classifier = nn.Sequential(
            nn.Linear(input_dim, input_dim // 2),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(input_dim // 2, input_dim // 4),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(input_dim // 4, num_answers)
        )
        
    def forward(self, fused_features):
        return self.answer_classifier(fused_features)
```

## 6. 통합 훈련 파이프라인

```python
class MultiModalTrainer:
    def __init__(self, model, config):
        self.model = model
        self.config = config
        
        # 옵티마이저
        self.optimizer = torch.optim.AdamW(
            model.parameters(),
            lr=config.learning_rate,
            weight_decay=config.weight_decay
        )
        
        # 스케줄러
        self.scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(
            self.optimizer, 
            T_max=config.num_epochs
        )
        
        # 손실 함수
        self.criterion = {
            'classification': nn.CrossEntropyLoss(),
            'captioning': nn.CrossEntropyLoss(ignore_index=0),
            'vqa': nn.CrossEntropyLoss()
        }
        
    def train_epoch(self, dataloader, task='classification'):
        self.model.train()
        total_loss = 0
        
        for batch in dataloader:
            images = batch['images']
            
            if task != 'classification':
                text = batch['text']
                labels = batch['labels']
                
                # 순전파
                outputs = self.model(images, text, task=task)
            else:
                labels = batch['labels']
                outputs = self.model(images, task=task)
            
            # 손실 계산
            loss = self.criterion[task](outputs, labels)
            
            # 역전파
            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()
            
            total_loss += loss.item()
            
        self.scheduler.step()
        return total_loss / len(dataloader)
    
    def evaluate(self, dataloader, task='classification'):
        self.model.eval()
        correct = 0
        total = 0
        
        with torch.no_grad():
            for batch in dataloader:
                images = batch['images']
                labels = batch['labels']
                
                if task != 'classification':
                    text = batch['text']
                    outputs = self.model(images, text, task=task)
                else:
                    outputs = self.model(images, task=task)
                
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()

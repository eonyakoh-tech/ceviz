# youtube_api.py
import requests
from googleapiclient.discovery import build

class YouTubeAnalyzer:
    def __init__(self, api_key):
        self.youtube = build('youtube', 'v3', developerKey=api_key)
    
    def get_video_info(self, video_id):
        request = self.youtube.videos().list(
            part="snippet,contentDetails,statistics",
            id=video_id
        )
        response = request.execute()
        return response['items'][0]
```

# transcript_extractor.py
from youtube_transcript_api import YouTubeTranscriptApi
import re

class TranscriptExtractor:
    def extract_transcript(self, video_url):
        video_id = self.extract_video_id(video_url)
        try:
            transcript = YouTubeTranscriptApi.get_transcript(video_id, languages=['ko', 'en'])
            return ' '.join([item['text'] for item in transcript])
        except Exception as e:
            return f"자막 추출 실패: {str(e)}"
    
    def extract_video_id(self, url):
        pattern = r'(?:v=|\/)([0-9A-Za-z_-]{11}).*'
        match = re.search(pattern, url)
        return match.group(1) if match else None
```

## 3. 통합 YouTube 분석기

```python
# youtube_research_agent.py
import yt_dlp
from youtube_transcript_api import YouTubeTranscriptApi
import json

class YouTubeResearchAgent:
    def __init__(self):
        self.ydl_opts = {
            'quiet': True,
            'no_warnings': True,
            'extract_flat': False,
        }
    
    def analyze_video(self, url):
        """YouTube 영상 종합 분석"""
        result = {
            'url': url,
            'metadata': {},
            'transcript': '',
            'summary': ''
        }
        
        # 1. 메타데이터 추출
        with yt_dlp.YoutubeDL(self.ydl_opts) as ydl:
            try:
                info = ydl.extract_info(url, download=False)
                result['metadata'] = {
                    'title': info.get('title'),
                    'description': info.get('description'),
                    'uploader': info.get('uploader'),
                    'duration': info.get('duration'),
                    'view_count': info.get('view_count'),
                    'upload_date': info.get('upload_date')
                }
            except Exception as e:
                result['error'] = f"메타데이터 추출 실패: {str(e)}"
        
        # 2. 자막 추출
        video_id = self.extract_video_id(url)
        if video_id:
            try:
                transcript = YouTubeTranscriptApi.get_transcript(
                    video_id, languages=['ko', 'en', 'auto']
                )
                result['transcript'] = ' '.join([item['text'] for item in transcript])
            except:
                result['transcript'] = "자막을 사용할 수 없습니다"
        
        return result
    
    def extract_video_id(self, url):
        """URL에서 video ID 추출"""
        import re
        patterns = [
            r'(?:v=|\/)([0-9A-Za-z_-]{11}).*',
            r'(?:embed\/)([0-9A-Za-z_-]{11})',
            r'(?:watch\?v=)([0-9A-Za-z_-]{11})'
        ]
        
        for pattern in patterns:
            match = re.search(pattern, url)
            if match:
                return match.group(1)
        return None
```

## 4. Obsidian Vault 연동

```python
# obsidian_integration.py
import os
from datetime import datetime

class ObsidianIntegrator:
    def __init__(self, vault_path):
        self.vault_path = vault_path
        self.youtube_folder = os.path.join(vault_path, "YouTube Research")
        
        if not os.path.exists(self.youtube_folder):
            os.makedirs(self.youtube_folder)
    
    def save_analysis(self, analysis_result):
        """YouTube 분석 결과를 Obsidian 노트로 저장"""
        title = analysis_result['metadata'].get('title', 'Unknown Video')
        safe_title = "".join(c for c in title if c.isalnum() or c in (' ', '-', '_')).rstrip()
        
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"{safe_title}_{timestamp}.md"
        filepath = os.path.join(self.youtube_folder, filename)
        
        content = self.generate_obsidian_note(analysis_result)
        
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(content)
        
        return filepath
    
    def generate_obsidian_note(self, analysis):
        """Obsidian 형식의 노트 생성"""
        metadata = analysis['metadata']
        
        content = f"""# {metadata.get('title', 'YouTube Video Analysis')}

## 📊 메타데이터
- **채널**: {metadata.get('uploader', 'Unknown')}
- **조회수**: {metadata.get('view_count', 'Unknown'):,}
- **업로드 날짜**: {metadata.get('upload_date', 'Unknown')}
- **영상 길이**: {metadata.get('duration', 'Unknown')} 초
- **URL**: {analysis['url']}

## 📝 설명
{metadata.get('description', 'No description available')[:500]}...

## 🎬 자막 내용
{analysis.get('transcript', 'No transcript available')}

## 🏷️ Tags
#youtube #research #video-analysis

---
*Created: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}*
"""
        return content
```

## 5. 메인 실행 파일

```python
# main.py
from youtube_research_agent import YouTubeResearchAgent
from obsidian_integration import ObsidianIntegrator

def main():
    # 설정
    OBSIDIAN_VAULT_PATH = "/path/to/your/obsidian/vault"
    
    # 에이전트 초기화
    youtube_agent = YouTubeResearchAgent()
    obsidian = ObsidianIntegrator(OBSIDIAN_VAULT_PATH)
    
    def analyze_youtube_video(url):
        """YouTube 영상 분석 및 Obsidian 저장"""
        print(f"🔍 YouTube 영상 분석 중: {url}")
        
        # 영상 분석
        analysis = youtube_agent.analyze_video(url)
        
        # Obsidian에 저장
        saved_path = obsidian.save_analysis(analysis)
        print(f"✅ 분석 완료! 저장 위치: {saved_path}")
        
        return analysis
    
    # 사용 예시
    video_url = "https://www.youtube.com/watch?v=VIDEO_ID"
    result = analyze_youtube_video(video_url)
    
if __name__ == "__main__":
    main()
```
